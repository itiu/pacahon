INPUT
{
  загрузить новую онтологию:

	аргумент [cmd:graph] -> указывает на корень загружаемого графа
	
	id-command:123457 
		cmd:command cmd:load-onto;
		cmd:from "sender-uuid";
		cmd:ticket "12345-67890-4322-132345-56754";
		cmd:args
			[
			onto:graph gost19:externalIdentifer;
			onto:graph "swrc:Organization"
			];
		
	gost19:externalIdentifer
      		rdf:type owl:DatatypeProperty ;
      		rdfs:subPropertyOf dc:identifier .

	swrc:Organization
      		rdfs:subClassOf
              		[ rdf:type owl:Restriction ;
                	owl:allValuesFrom xsd:string ;
                	owl:onProperty swrc:name
              		] .
}
OUTPUT
{
	id-command:123457 cmd:result "Ok".
}
